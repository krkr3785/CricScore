Connect CricScore
Connect your free Supabase database to sync data across all your devices.
Quick Setup (2 min):
1. Go to _sbClient.com → New Project (free)
2. Go to Project Settings → API
3. Copy Project URL and anon public key below
4. Go to SQL Editor and run the setup SQL ↓
-- Run this in Supabase SQL Editor:
create table if not exists cw_profiles (email text primary key, name text);
create table if not exists cw_data (user_email text primary key, data jsonb, updated_at timestamptz);
alter table cw_profiles enable row level security;
alter table cw_data enable row level security;
create policy "users manage own profile" on cw_profiles for all using (true) with check (true);
create policy "users manage own data" on cw_data for all using (true) with check (true);
Use locally without cloud sync
